62 /******************************************************************************
63  * Ctrl-C destructor. Tries to quit gracefully with the first Ctrl-C
64  * press, then just exits.
65  *****************************************************************************/
66 static void quit(int /*sig*/)
67 {
68  static int kHit = 0;
69  runLoop = false;
70  fprintf(stderr, "Trying to exit gracefully - %dst attempt\n", kHit + 1);
71  kHit++;
72 
73  if (kHit == 2) {
74  fprintf(stderr, "Exiting without cleanup\n");
75  exit(0);
76  }
77 }
