83 void ISSBoxLifting::applyInitialState(const MatNd* initialState)
84 {
85  // Set the position to the box' rigid body joints directly in global world coordinates
86  bool b0 = RcsGraph_setJoint(graph, "DofBaseX", initialState->ele[0]);
87  bool b1 = RcsGraph_setJoint(graph, "DofBaseY", initialState->ele[1]);
88  bool b2 = RcsGraph_setJoint(graph, "DofBaseThZ", initialState->ele[2]);
89  bool b3 = RcsGraph_setJoint(graph, "DofChestZ", initialState->ele[3]);
90  bool b4 = RcsGraph_setJoint(graph, "lbr_joint_2_L", initialState->ele[4]);
91  bool b5 = RcsGraph_setJoint(graph, "lbr_joint_2_R", initialState->ele[5]);
92  if (!(b0 && b1 && b2 && b3 && b4 && b5)) {
93  throw std::invalid_argument("Setting graph failed for at least one of the joints!");
94  }
95 
96  // Update the forward kinematics
97  RcsGraph_setState(graph, graph->q, graph->q_dot);
98 }
