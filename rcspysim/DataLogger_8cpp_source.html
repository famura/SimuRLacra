57 void DataLogger::start(
58  const BoxSpace* observationSpace,
59  const BoxSpace* actionSpace,
60  unsigned int maxStepCount,
61  const char* filename)
62 {
63  // Guard against concurrency
64  std::unique_lock<std::recursive_mutex> lock(mutex);
65  if (running) {
66  RLOG(1, "Already running!");
67  return;
68  }
69  running = true;
70  // Determine filename
71  std::string fname;
72  if (filename == nullptr) {
73  // Generate
74  std::ostringstream os;
75  os << baseFileName << (fileCounter++) << ".csv";
76  fname = os.str();
77  }
78  else {
79  fname = filename;
80  }
81  // Open output file
82  output.open(fname);
83 
84  // Write header (column names)
85  output << R"("steps)";
86  for (auto& name : observationSpace->getNames()) {
87  output << "\",\"" << name;
88  }
89  for (auto& name : actionSpace->getNames()) {
90  output << "\",\"" << name;
91  }
92  output << "\"" << std::endl;
93 
94  // Allocate buffer
95  buffer = MatNd_create(maxStepCount, observationSpace->getNames().size() + actionSpace->getNames().size());
96  currentStep = 0;
97 
98  RLOG(0, "Logging started!");
99 }

101 void DataLogger::stop()
102 {
103  // Guard against concurrency
104  std::unique_lock<std::recursive_mutex> lock(mutex);
105  if (!running) {
106  return;
107  }
108  running = false;
109 
110  // Write buffer contents to csv
111  for (unsigned int row = 0; row < currentStep; ++row) {
112 
113  // Write step number
114  output << row;
115 
116  // Write elements
117  for (unsigned int col = 0; col < buffer->n; ++col) {
118  output << "," << MatNd_get2(buffer, row, col);
119  }
120  output << std::endl;
121  }
122 
123  // Close file
124  output.flush();
125  output.close();
126  output.clear();
127 
128  // Delete buffer
129  MatNd_destroy(buffer);
130 
131  RLOG(0, "Logging stopped!");
132 }

134 void DataLogger::record(const MatNd* observation, const MatNd* action)
135 {
136  // Try to obtain lock. If it's blocked, it's blocked by start or stop, so we don't want to lock anyways.
137  std::unique_lock<std::recursive_mutex> lock(mutex, std::try_to_lock);
138  if (!lock.owns_lock() || !running) {
139  return;
140  }
141 
142  // Add a line to buffer
143  if (currentStep >= buffer->m) {
144  stop();
145  return;
146  }
147  double* lineBuffer = MatNd_getRowPtr(buffer, currentStep++);
148 
149  VecNd_copy(&lineBuffer[0], observation->ele, observation->m);
150  VecNd_copy(&lineBuffer[observation->m], action->ele, action->m);
151 }
