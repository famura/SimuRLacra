67 BodyParamInfo* PhysicsParameterManager::getBodyInfo(const char* bodyName)
68 {
69  // check if used already
70  for (BodyParamInfo& existing : bodyInfos) {
71  if (STREQ(existing.body->name, bodyName)) {
72  return &existing;
73  }
74  }
75  // not found, so add
76  bodyInfos.emplace_back(graph, bodyName, physicsConfig);
77  return &bodyInfos.back();
78 }
