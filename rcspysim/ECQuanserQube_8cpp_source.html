70  virtual ActionModel* createActionModel()
71  {
72  std::string actionModelType = "unspecified";
73  properties->getProperty(actionModelType, "actionModelType");
74 
75  if (actionModelType == "joint_acc") {
76  /** V_m_max = 5 V and theta_dot_0 = 0 rad/s
77  * tau_max = k_m (V_m_max - k_m * theta_dot_0) / R_m = 0.04375 Nm
78  * J = m * l^2 / 12 = 0.000033282 kg m^2
79  * alpha_ddot_max = tau_max / J = 1314.52 rad/s^2 */
80  double max_action = 1314.52/10.; // divide by 10 since if seem unrealistic otherwise
81  properties->getProperty(max_action, "maxAction");
82  // By integrating the position command twice before applying it as action we command the acceleration
83  return new AMIntegrate2ndOrder(new AMJointControlPosition(graph), max_action);
84  // return new AMJointControlAcceleration(graph); // not working
85  }
86 
87  else {
88  std::ostringstream os;
89  os << "Unsupported action model type: " << actionModelType;
90  throw std::invalid_argument(os.str());
91  }
92  }

133  void getHUDText(
134  std::vector<std::string>& linesOut, double currentTime, const MatNd* obs, const MatNd* currentAction,
135  PhysicsBase* simulator, PhysicsParameterManager* physicsManager, ForceDisturber* forceDisturber) override
136  {
137  // Obtain simulator name
138  const char* simName = "None";
139  if (simulator != NULL) {
140  simName = simulator->getClassName();
141  }
142 
143  // common stuff
144  linesOut.emplace_back(
145  string_format("physics engine: %s sim time: %2.3f s",
146  simName, currentTime));
147 
148  unsigned int numPosCtrlJoints = 0;
149  unsigned int numTrqCtrlJoints = 0;
150  // Iterate over unconstrained joints
151  RCSGRAPH_TRAVERSE_JOINTS(graph) {
152  if (JNT->jacobiIndex != -1) {
153  if (JNT->ctrlType == RCSJOINT_CTRL_TYPE::RCSJOINT_CTRL_POSITION) {
154  numPosCtrlJoints++;
155  }
156  else if (JNT->ctrlType == RCSJOINT_CTRL_TYPE::RCSJOINT_CTRL_TORQUE) {
157  numTrqCtrlJoints++;
158  }
159  }
160  }
161  linesOut.emplace_back(
162  string_format("num joints: %d total, %d pos ctrl, %d trq ctrl", graph->nJ, numPosCtrlJoints,
163  numTrqCtrlJoints));
164 
165  linesOut.emplace_back(
166  string_format(" pole angles: [% 3.2f,% 3.2f] deg",
167  RCS_RAD2DEG(obs->ele[0]), RCS_RAD2DEG(obs->ele[1])));
168 
169  const double* distForce = forceDisturber->getLastForce();
170  linesOut.emplace_back(
171  string_format("disturbances: [% 3.1f,% 3.1f,% 3.1f] N",
172  distForce[0], distForce[1], distForce[2]));
173 
174  std::stringstream ss;
175  ss << "actions: [";
176  for (unsigned int i = 0; i < currentAction->m - 1; i++) {
177  ss << std::fixed << std::setprecision(3) << MatNd_get(currentAction, i, 0) << ", ";
178  if (i == 6) {
179  ss << "\n ";
180  }
181  }
182  ss << std::fixed << std::setprecision(3) << MatNd_get(currentAction, currentAction->m - 1, 0) << "]";
183  linesOut.emplace_back(string_format(ss.str()));
184 
185  if (physicsManager != NULL) {
186  // Include physics parameters
187 
188  // Get bodies from graph
189  RcsBody* arm = RcsGraph_getBodyByName(graph, "Arm");
190  RcsBody* pendulum = RcsGraph_getBodyByName(graph, "Pendulum");
191 
192 
193  linesOut.emplace_back(string_format(
194  "arm mass: %1.3f kg pendulum mass: %1.3f kg",
195  arm->m, pendulum->m));
196 
197  linesOut.emplace_back(string_format(
198  "arm length: %1.3f m pendulum length: %1.3f m",
199  arm->shape[0]->extents[2], pendulum->shape[0]->extents[2]));
200 
201  linesOut.emplace_back(string_format(
202  "arm radius: %1.3f m pendulum radius: %1.3f m",
203  arm->shape[0]->extents[0], pendulum->shape[0]->extents[0]));
204  }
205  }
