58 /******************************************************************************
59  * Ctrl-C destructor. Tries to quit gracefully with the first Ctrl-C
60  * press, then just exits.
61  *****************************************************************************/
62 static void quit(int /*sig*/)
63 {
64  static int kHit = 0;
65  runLoop = false;
66  fprintf(stderr, "Trying to exit gracefully - %dst attempt\n", kHit + 1);
67  kHit++;
68 
69  if (kHit == 2) {
70  fprintf(stderr, "Exiting without cleanup\n");
71  exit(0);
72  }
73 }
