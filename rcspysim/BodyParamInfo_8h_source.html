43 /**
44  * Information on a body and changing physics params.
45  *
46  * Apart from the body, this struct holds material parameters not storable in RcsBody, and flags to track modified parameters.
47  */
48 struct BodyParamInfo
49 {
50  enum ModifiedFlag
51  {
52  // body mass changed
53  MOD_MASS = 1 << 0,
54  // body center of gravity changed
55  MOD_COM = 1 << 1,
56  // body inertia changed
57  MOD_INERTIA = 1 << 2,
58  // collision shapes changed
59  MOD_SHAPE = 1 << 3,
60  // position changed
61  MOD_POSITION = 1 << 4
62  };
63 
64  // The graph containing the body
65  RcsGraph* graph;
66 
67  // The body
68  RcsBody* body;
69 
70  // prefix for parameter names
71  std::string paramNamePrefix;
72 
73  // body material - this is the material of the body's first shape.
74  PhysicsMaterial material;
75 
76  // flags tracking the modified state of the body
77  int modifiedFlag;
78 
79  BodyParamInfo(RcsGraph* graph, const char* bodyName, PhysicsConfig* physicsConfig);
80 
81  // reset all change flags
82  void resetChanged();
83 
84  // test if a parameter changed
85  inline bool isChanged(int flag)
86  {
87  return (modifiedFlag & flag) == flag;
88  }
89 
90 
91  // mark if a parameter as changed
92  inline void markChanged(int flag)
93  {
94  modifiedFlag |= flag;
95  }
96 };
