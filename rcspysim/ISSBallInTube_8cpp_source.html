75 void ISSBallInTube::applyInitialState(const MatNd* initialState)
76 {
77  // Set the position to the box' rigid body joints directly in global world coordinates
78  bool b0 = RcsGraph_setJoint(graph, "DofBaseX", initialState->ele[0]);
79  bool b1 = RcsGraph_setJoint(graph, "DofBaseY", initialState->ele[1]);
80  bool b2 = RcsGraph_setJoint(graph, "DofBaseThZ", initialState->ele[2]);
81  bool b3 = RcsGraph_setJoint(graph, "DofChestZ", initialState->ele[3]);
82  if (!(b0 && b1 && b2 && b3)) {
83  throw std::invalid_argument("Setting graph failed for at least one of the joints!");
84  }
85 
86  // Update the forward kinematics
87  RcsGraph_setState(graph, graph->q, graph->q_dot);
88 }
