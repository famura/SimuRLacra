49 /*! A simple time-based policy
50  * Yields 2 actions: the first one is oscillating round 0 with am amplitude of 0.2 and the second one is constant 0.
51  */
52 class SimpleControlPolicy : public ControlPolicy
53 {
54 private:
55  //! Internal clock
56  double t;
57 
58 public:
59  SimpleControlPolicy() { t = 0; }
60 
61  virtual void computeAction(MatNd* action, const MatNd* observation)
62  {
63  action->ele[0] = 0.2*std::cos(2.*M_PI*t)*(135*M_PI/180);
64  action->ele[1] = 0.0;
65  t += 0.01;
66  }
67 };

71 RcsPyBot::RcsPyBot(PropertySource* propertySource)
72 {
73  // Load experiment config
74  config = ExperimentConfig::create(propertySource);
75 
76  // Check if all joints are position controlled for skipping a later inverse dynamics control (compliance)
77  unsigned int numPosCtrlJoints = 0;
78  RCSGRAPH_TRAVERSE_JOINTS(config->graph) {
79  if (JNT->jacobiIndex != -1) {
80  if (JNT->ctrlType == RCSJOINT_CTRL_TYPE::RCSJOINT_CTRL_POSITION) {
81  numPosCtrlJoints++;
82  }
83  }
84  }
85  allJointsPosCtrl = config->graph->nJ == numPosCtrlJoints;
86 
87  // Set MotionControlLayer members
88  currentGraph = config->graph;
89  desiredGraph = RcsGraph_clone(currentGraph);
90 
91  // Control policy is set later
92  controlPolicy = NULL;
93 
94  // Init temp matrices, making sure the initial command is identical to the initial state
95  q_ctrl = MatNd_clone(desiredGraph->q);
96  qd_ctrl = MatNd_clone(desiredGraph->q_dot);
97  T_ctrl = MatNd_create(desiredGraph->dof, 1);
98 
99  observation = config->observationModel->getSpace()->createValueMatrix();
100  action = config->actionModel->getSpace()->createValueMatrix();
101 
102  // ActionModel and observationModel expect a reset() call before they are used
103  config->actionModel->reset();
104  config->observationModel->reset();
105 }

107 RcsPyBot::~RcsPyBot()
108 {
109  // Delete temporary matrices
110  MatNd_destroy(q_ctrl);
111  MatNd_destroy(qd_ctrl);
112  MatNd_destroy(T_ctrl);
113 
114  MatNd_destroy(observation);
115  MatNd_destroy(action);
116 
117  // The hardware components also use currentGraph, so it may only be destroyed by the MotionControlLayer destructor.
118  // however, currentGraph is identical to config->graph, which is owned.
119  // to solve this, set config->graph to NULL.
120  config->graph = NULL;
121  // Also, desiredGraph is a clone, so it must be destroyed. Can't set MotionControlLayer::ownsDesiredGraph = true
122  // since it's private, so do it manually here.
123  RcsGraph_destroy(desiredGraph);
124 
125  // Delete experiment config
126  delete config;
127 }

129 void RcsPyBot::setControlPolicy(ControlPolicy* controlPolicy)
130 {
131  std::unique_lock<std::mutex> lock(controlPolicyMutex);
132  this->controlPolicy = controlPolicy;
133  if (controlPolicy == NULL) {
134  // Command initial state
135  RcsGraph_getDefaultState(desiredGraph, q_ctrl);
136  MatNd_setZero(qd_ctrl);
137  MatNd_setZero(T_ctrl);
138  }
139  // Reset model states
140  config->observationModel->reset();
141  config->actionModel->reset();
142 }

144 void RcsPyBot::updateControl()
145 {
146  // Aggressive locking here is ok, setControlPolicy doesn't take long
147  std::unique_lock<std::mutex> lock(controlPolicyMutex);
148 
149  // Read observation from current graph
150  config->observationModel->computeObservation(observation, action, config->dt);
151 
152  // Compute action
153  if (controlPolicy != NULL) {
154  controlPolicy->computeAction(action, observation);
155 
156  // Run action through action model
157  config->actionModel->computeCommand(q_ctrl, qd_ctrl, T_ctrl, action, getCallbackUpdatePeriod());
158  }
159 
160  // Inverse dynamics in joint space (compliance control)
161  if (!allJointsPosCtrl) {
162  Rcs::ControllerBase::computeInvDynJointSpace(T_ctrl, config->graph, q_ctrl, 1000.);
163  }
164 
165  // Update desired state graph
166 
167  // TODO if(writeCommands), but seriously, shouldn't the component handle that stuff itself?
168  // Command action to hardware (and update desiredGraph)
169  setMotorCommand(q_ctrl, qd_ctrl, T_ctrl);
170 
171  // Can unlock now, lock only guards controlPolicy and ctrl vectors
172  lock.unlock();
173 
174  // Log data
175  double reward = 0.0; // so far, the reward is only computed on the python side
176  logger.record(observation, action, reward);
177 }
