94  bool noEventGui = argP.hasArgument("-noEventGui", "Don't launch EventGui");
95  bool seqSim = argP.hasArgument("-sequentialPhysics", "Physics simulation step"
96  "in updateGraph()");
97  bool seqViewer = argP.hasArgument("-sequentialViewer", "Viewer frame call "
98  "in \"Render\" event");
99  bool pause = argP.hasArgument("-pause", "Pause after each process() call");
100  bool sync = argP.hasArgument("-sync", "Run as sequential as possible");

106 
107  if (sync) {
108  seqSim = true;
109  seqViewer = true;
110  }
111 
112  EntityBase entity;
113  if (pause) {
114  entity.call("TogglePause");

116 
117  // Register lifecycle events
118  auto updateGraph = entity.registerEvent<RcsGraph*>("UpdateGraph");

135 // policy.setSpeedLimitCheck(!noSpeedCheck);
136  policy.setJointLimitCheck(!noJointCheck);
137  policy.setCollisionCheck(!noCollCheck);
138  RcsGraph* graph = policy.getExperiment()->graph;
139 
140  // setup viewer and graph components
141  GraphicsWindow viewer(&entity, false, seqViewer, simpleGraphics);
142  GraphComponent graphC(&entity, graph);
143 
144  // setup hardware
145  std::vector<ComponentBase*> hwc = getHardwareComponents(entity, graph);
146 
147  if (hwc.empty()) {
148  // no hardware set, use physics simulator
149 // RCSGRAPH_TRAVERSE_JOINTS(graph) {
150 // JNT->ctrlType = RCSJOINT_CTRL_POSITION;
151 // }
152 
153  auto pc = new PhysicsComponent(&entity, graph, physicsEngine,
154  physicsCfg, !seqSim);
155 
156  if (seqSim && seqViewer) {
157 #if defined (USE_BULLET)
158  osg::ref_ptr<BulletDebugNode> btNd =
159  new BulletDebugNode(pc->getPhysicsSimulation(), viewer.getFrameMtx());
160  viewer.add(btNd.get());
161 #endif
162  }
163 
164  hwc.push_back(pc);
165  }
166 
167  // Setup viewer key callbacks
168  viewer.setKeyCallback('d', [&entity](char k) {
169  RLOG(0, "Reseting IME failure flag");
170  entity.publish("ResetErrorFlag");
171  }, "Reseting IME failure flag");
172 
173  viewer.setKeyCallback('m', [&entity, ttc](char k) {
174  RLOG(0, "Move robot to initial pose");
175  entity.publish("InitializeMovement", ttc);
176  }, "Move robot to initial pose");
177 
178  // End of configuration, print help if requested
179  if (argP.hasArgument("-h")) {
180  entity.print(std::cout);
181  return;
182  }
183 
184  // Setup event GUI if requested, only do so now since all events are registered
185  if (!noEventGui) {
186  EventGui::create(&entity);
187  }
188 
189 
190  // Start threads (if any)
191  RPAUSE_MSG_DL(1, "Start");
192  entity.publish("Start");

195  nIter, entity.queueSize());
196 
197  // Initialization sequence
198  entity.publish("Render");
199  nIter = entity.processUntilEmpty(10);
200  RLOG(1, "Render took %d process() calls, queue is %zu",
201  nIter, entity.queueSize());
202 
203  RPAUSE_MSG_DL(1, "updateGraph");
204  updateGraph->call(graphC.getGraph());
205  nIter = entity.processUntilEmpty(10);
206  RLOG(1, "updateGraph took %d process() calls, queue is %zu",
207  nIter, entity.queueSize());
208 
209  updatePolicy->call(graphC.getGraph());
210  entity.processUntilEmpty(10);
211  entity.publish("Render");
212  entity.processUntilEmpty(10);
213 
214  RPAUSE_MSG_DL(1, "updateGraph");
215  updateGraph->call(graphC.getGraph());
216  nIter = entity.processUntilEmpty(10);
217  RLOG(1, "updateGraph took %d process() calls, queue is %zu",
218  nIter, entity.queueSize());
219 
220  updatePolicy->call(graphC.getGraph());
221  entity.processUntilEmpty(10);
222 
223  entity.publish<const RcsGraph*>("InitFromState", graphC.getGraph());
224  entity.publish("Render");
225  entity.processUntilEmpty(10);
226

239 
240  RPAUSE_MSG_DL(1, "EnableCommands");
241  entity.publish("EnableCommands");
242  nIter = entity.processUntilEmpty(10);
243  RLOG(1, "InitFromState++ took %d process() calls, queue is %zu",
244  nIter, entity.queueSize());
245  RPAUSE_MSG_DL(1, "Enter runLoop");
246 
247 
248  while (runLoop) {
249  double dtProcess = Timer_getSystemTime();
250  updateGraph->call(graphC.getGraph());
251

255  postUpdateGraph->call(policy.getDesiredGraph(), graphC.getGraph());
256 
257  // compute input from policy
258  updatePolicy->call(graphC.getGraph());
259 
260  // This is called right before sending the commands
261  // checkEmergencyStop->call();

265 
266 // setComplianceCommand->call(dmpc.getComplianceWrench());

273  entity.process();

276  dtProcess = Timer_getSystemTime() - dtProcess;
277 
278 
279  char text[256];
280  snprintf(text, 256, "Time: %.3f dt: %.1f msec (< %.1f msec) queue: %zu",
281  entity.getTime(), 1.0e3*dtProcess, 1.0e3*entity.getDt(), queueSize);
282  entity.publish("SetTextLine", std::string(text), 1);
283 
284  entity.publish("SetTextLine", policy.getStateText(), 2);

288  NULL, NULL, NULL);

290  entity.publish("SetTextLine", hudLines[i], (int) i + 3);
291  }

304 
305  loopCount++;

310  }

312  entity.publish<>("Stop");
