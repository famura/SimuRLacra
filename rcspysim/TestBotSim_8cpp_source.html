61 /******************************************************************************
62  * Ctrl-C destructor. Tries to quit gracefully with the first Ctrl-C
63  * press, then just exits.
64  *****************************************************************************/
65 static void quit(int /*sig*/)
66 {
67  static int kHit = 0;
68  runLoop = false;
69  fprintf(stderr, "Trying to exit gracefully - %dst attempt\n", kHit + 1);
70  kHit++;
71 
72  if (kHit == 2) {
73  fprintf(stderr, "Exiting without cleanup\n");
74  exit(0);
75  }
76 }

78 int main(int argc, char** argv)
79 {
80  RMSG("Starting Rcs...");
81 
82  Rcs::KeyCatcherBase::registerKey("q", "Quit");
83  Rcs::KeyCatcherBase::registerKey("l", "Start/stop data logging");
84  Rcs::KeyCatcherBase::registerKey("p", "Activate control policy");
85  Rcs::KeyCatcherBase::registerKey("o", "Deactivate policy and return to initial state");
86 
87  // Ctrl-C callback handler
88  signal(SIGINT, quit);
89 
90  // This initialize the xml library and check potential mismatches between the version it was compiled for and
91  // the actual shared library used.
92  LIBXML_TEST_VERSION;
93 
94  // Parse command line arguments
95  Rcs::CmdLineParser argP(argc, argv);
96  char xmlFileName[128] = "ex_p3l-ika_export.xml";
97  char directory[128] = "../config/ENVIRONMENT_FOLDER";
98  argP.getArgument("-dl", &RcsLogLevel, "Debug level (default is 0)");
99  argP.getArgument("-f", xmlFileName, "Configuration file name");
100  argP.getArgument("-dir", directory, "Configuration file directory");
101  bool valgrind = argP.hasArgument("-valgrind", "Start without GUIs and graphics");
102 // bool simpleGraphics = argP.hasArgument("-simpleGraphics", "OpenGL without fancy stuff (shadows, anti-aliasing)");
103 
104  const char* hgr = getenv("SIT");
105  if (hgr != NULL) {
106  std::string meshDir = std::string(hgr) + std::string("/Data/RobotMeshes/1.0/data");
107  Rcs_addResourcePath(meshDir.c_str());
108  }
109 
110 // Rcs_addResourcePath("../config");
111  Rcs_addResourcePath(directory);
112 
113  // show help if requested
114  if (argP.hasArgument("-h", "Show help message")) {
115  Rcs::KeyCatcherBase::printRegisteredKeys();
116  Rcs::CmdLineParser::print();
117  Rcs_printResourcePath();
118  return 0;
119  }
120 
121  // Create simulated robot from config file
122  RMSG("Creating robot...");
123  Rcs::RcsPyBot bot(new Rcs::PropertySourceXml(xmlFileName));
124 
125  // TODO add hardware components (which I don't have currently)
126  // Add physics simulator for testing
127  Rcs::PhysicsParameterManager* ppmanager = bot.getConfig()->createPhysicsParameterManager();
128  Rcs::PhysicsBase* simImpl = ppmanager->createSimulator(bot.getConfig()->properties->getChild("initDomainParam"));
129 
130  bot.getConfig()->actionModel->reset();
131  bot.getConfig()->observationModel->reset();
132 
133  Rcs::PhysicsSimulationComponent* sim = new Rcs::PhysicsSimulationComponent(simImpl);
134  sim->setUpdateFrequency(1.0/bot.getConfig()->dt);
135  bot.addHardwareComponent(sim);
136  bot.setCallbackTriggerComponent(sim); // and it does drive the update loop
137 
138  // Add viewer component
139 #ifdef GRAPHICS_AVAILABLE
140  Rcs::ViewerComponent* vc = NULL;
141  Rcs::HUD* hud = NULL;
142  if (!valgrind) {
143  vc = new Rcs::ViewerComponent(bot.getGraph(), bot.getCurrentGraph(), true);
144  hud = new Rcs::HUD(0, 0, 1024, 140);
145  vc->getViewer()->add(hud);
146 
147  bot.getConfig()->initViewer(vc->getViewer());
148  bot.addHardwareComponent(vc);
149  }
150 #endif
151 
152  // Load control policy
153  Rcs::ControlPolicy* controlPolicy = NULL;
154  auto policyConfig = bot.getConfig()->properties->getChild("policy");
155  if (policyConfig->exists()) {
156  controlPolicy = Rcs::ControlPolicy::create(policyConfig);
157  REXEC(1) {
158  std::cout << "Loaded policy specified in the config file." << std::endl;
159  }
160  }
161  else {
162  REXEC(1) {
163  std::cout << "Could not load a policy!" << std::endl;
164  }
165  }
166 
167 // controlPolicy = Rcs::loadMLPPolicyFromXml(policyFile.c_str(),
168 // bot.getConfig()->observationModel->getDim(), bot.getConfig()
169 // ->actionModel->getDim());
170 
171  // Start
172  bot.startThreads();
173  RMSG("Started robot.");
174  bool startLoggerNextPolicyStart = false;
175 
176  // Main loop
177  runLoop = true;
178  RMSG("Main loop is running ...");
179  while (runLoop) {
180  // Check keys
181 #ifdef GRAPHICS_AVAILABLE
182  if (vc && vc->getKeyCatcher()->getAndResetKey('q')) {
183  runLoop = false;
184  }
185  if (vc && vc->getKeyCatcher()->getAndResetKey('l')) {
186  if (bot.logger.isRunning()) {
187  bot.logger.stop();
188  }
189  else if (bot.getControlPolicy() != controlPolicy) {
190  // Defer until policy start
191  startLoggerNextPolicyStart = true;
192  RMSG("Deferring logger start until the policy is activated.");
193  }
194  else {
195  bot.logger.start(bot.getConfig()->observationModel->getSpace(),
196  bot.getConfig()->actionModel->getSpace(), 500);
197  }
198  }
199  if (vc && vc->getKeyCatcher()->getAndResetKey('p')) {
200  if (startLoggerNextPolicyStart) {
201  bot.logger.start(bot.getConfig()->observationModel->getSpace(),
202  bot.getConfig()->actionModel->getSpace(), 500);
203  }
204  bot.setControlPolicy(controlPolicy);
205  RMSG("Control policy active.");
206  }
207  if (vc && vc->getKeyCatcher()->getAndResetKey('o')) {
208  bot.setControlPolicy(NULL);
209  RMSG("Holding initial state.");
210  }
211 
212  if (hud) {
213  auto hudText = bot.getConfig()->getHUDText(
214  simImpl->time(), bot.getObservation(), bot.getAction(), simImpl, ppmanager, nullptr);
215  hud->setText(hudText);
216  }
217 #endif
218 
219  // Wait a bit till next update
220  Timer_waitDT(0.01);
221  }
222 
223  // Terminate
224  RMSG("Terminating...");
225  bot.stopThreads();
226  bot.disconnectCallback();
227 
228  delete controlPolicy;
229  delete ppmanager;
230 
231  // Clean up global stuff. From the libxml2 documentation:
232  // WARNING: if your application is multithreaded or has plugin support
233  // calling this may crash the application if another thread or a plugin is
234  // still using libxml2. It's sometimes very hard to guess if libxml2 is in
235  // use in the application, some libraries or plugins may use it without
236  // notice. In case of doubt abstain from calling this function or do it just
237  // before calling exit() to avoid leak reports from valgrind !
238  xmlCleanupParser();
239 
240  fprintf(stderr, "Thanks for using the Rcs libraries\n");
241  return 0;
242 }
