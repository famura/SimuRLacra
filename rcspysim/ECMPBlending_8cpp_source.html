72  virtual ActionModel* createActionModel()
73  {
74  std::string actionModelType = "unspecified";
75  properties->getProperty(actionModelType, "actionModelType");
76 
77  // Get the method how to combine the movement primitives / tasks given their activation
78  std::string taskCombinationMethod = "unspecified";
79  properties->getProperty(taskCombinationMethod, "taskCombinationMethod");
80  TaskCombinationMethod tcm = AMDynamicalSystemActivation::checkTaskCombinationMethod(taskCombinationMethod);
81 
82  RcsBody* effector = RcsGraph_getBodyByName(graph, "Effector");
83  RCHECK(effector);
84 
85  if (actionModelType == "ds_activation") {
86  // Obtain the inner action model
87  std::unique_ptr<AMIKGeneric> innerAM(new AMIKGeneric(graph));
88 
89  // Check if the MPs are defined on position or velocity level
90  if (properties->getPropertyBool("positionTasks", true)) {
91  innerAM->addTask(new TaskPosition1D("X", graph, effector, nullptr, nullptr));
92  innerAM->addTask(new TaskPosition1D("Y", graph, effector, nullptr, nullptr));
93  }
94  else {
95  innerAM->addTask(new TaskVelocity1D("Xd", graph, effector, nullptr, nullptr));
96  innerAM->addTask(new TaskVelocity1D("Yd", graph, effector, nullptr, nullptr));
97  }
98 
99  // Obtain the task data
100  auto& taskSpec = properties->getChildList("tasks");
101  if (taskSpec.empty()) {
102  throw std::invalid_argument("No tasks specified!");
103  }
104  std::vector<std::unique_ptr<DynamicalSystem>> tasks;
105  for (auto ts : taskSpec) {
106  // All tasks cover the x and y coordinate
107  tasks.emplace_back(DynamicalSystem::create(ts, innerAM->getDim()));
108  }
109 
110  // Setup task-based action model
111  std::vector<DynamicalSystem*> taskRel;
112  for (auto& task : tasks) {
113  taskRel.push_back(task.release());
114  }
115 
116  // Create the action model
117  return new AMDynamicalSystemActivation(innerAM.release(), taskRel, tcm);
118  }
119 
120  else if (actionModelType == "ik_activation") {
121  // Create the action model
122  auto amIK = new AMIKControllerActivation(graph, tcm);
123  std::vector<Task*> tasks;
124 
125  // Check if the tasks are defined on position or velocity level. Adapt their parameters if desired.
126  if (properties->getPropertyBool("positionTasks", true)) {
127  RcsBody* goalLL = RcsGraph_getBodyByName(graph, "GoalLL");
128  RcsBody* goalUL = RcsGraph_getBodyByName(graph, "GoalUL");
129  RcsBody* goalLR = RcsGraph_getBodyByName(graph, "GoalLR");
130  RcsBody* goalUR = RcsGraph_getBodyByName(graph, "GoalUR");
131  RCHECK(goalLL);
132  RCHECK(goalUL);
133  RCHECK(goalLR);
134  RCHECK(goalUR);
135  int i = 0;
136 
137  tasks.emplace_back(new TaskPosition3D(graph, effector, goalLL, nullptr));
138  tasks.emplace_back(new TaskPosition3D(graph, effector, goalUL, nullptr));
139  tasks.emplace_back(new TaskPosition3D(graph, effector, goalLR, nullptr));
140  tasks.emplace_back(new TaskPosition3D(graph, effector, goalUR, nullptr));
141  for (auto task : tasks) {
142  std::stringstream taskName;
143  taskName << " Position " << i++ << " [m]";
144  task->resetParameter(
145  Task::Parameters(-1.2, 1.2, 1.0, static_cast<std::string>("X") + taskName.str()));
146  task->addParameter(
147  Task::Parameters(-1.2, 1.2, 1.0, static_cast<std::string>("Y") + taskName.str()));
148  task->addParameter(Task::Parameters(0.1, 0.2, 1.0, static_cast<std::string>("Z") + taskName.str()));
149  }
150  }
151  else {
152  throw std::invalid_argument("The combination of velocity-based tasks and AMIKControllerActivation is"
153  "not supported!");
154  }
155 
156  // Add the tasks
157  for (auto t : tasks) { amIK->addTask(t); }
158 
159  // Set the tasks' desired states
160  std::vector<PropertySource*> taskSpec = properties->getChildList("taskSpecIK");
161  amIK->setXdesFromTaskSpec(taskSpec);
162 
163  return amIK;
164  }
165 
166  else {
167  std::ostringstream os;
168  os << "Unsupported action model type: " << actionModelType;
169  throw std::invalid_argument(os.str());
170  }
171  }

264  BodyParamInfo* eff_bpi = physicsManager->getBodyInfo("Effector");
