73  virtual ActionModel* createActionModel()
74  {
75  std::string actionModelType = "unspecified";
76  properties->getProperty(actionModelType, "actionModelType");
77 
78  if (actionModelType == "joint_pos") {
79  return new AMJointControlPosition(graph);
80  }
81 
82  else if (actionModelType == "joint_acc") {
83  double maxAction = 120*M_PI/180; // [1/s^2]
84  properties->getProperty(maxAction, "maxAction");
85  return new AMIntegrate2ndOrder(new AMJointControlPosition(graph), maxAction);
86  }
87 
88  else if (actionModelType == "plate_angpos") {
89  return new AMPlateAngPos(graph);
90  }
91 
92  else if (actionModelType == "plate_angvel") {
93  double maxAction = 120*M_PI/180; // [1/s]
94  properties->getProperty(maxAction, "maxAction");
95  return new AMIntegrate1stOrder(new AMPlateAngPos(graph), maxAction);
96  }
97 
98  else if (actionModelType == "plate_angacc") {
99  double maxAction = 120*M_PI/180; // [1/s^2]
100  properties->getProperty(maxAction, "maxAction");
101  return new AMIntegrate2ndOrder(new AMPlateAngPos(graph), maxAction);
102  }
103 
104  else if (actionModelType == "plate_acc5d") {
105  MatNd* maxAction;
106  MatNd_fromStack(maxAction, 5, 1);
107  // Use different max action for linear/angular
108  double max_lin = 0.5; // [m/s^2]
109  double max_ang = 120*M_PI/180; // [1/s^2]
110 
111  maxAction->ele[0] = max_lin;
112  maxAction->ele[1] = max_lin;
113  maxAction->ele[2] = max_lin;
114  maxAction->ele[3] = max_ang;
115  maxAction->ele[4] = max_ang;
116 
117  return new AMIntegrate2ndOrder(new AMPlatePos5D(graph), maxAction);
118  }
119 
120  else {
121  std::ostringstream os;
122  os << "Unsupported action model type: " << actionModelType;
123  throw std::invalid_argument(os.str());
124  }
125  }

127  virtual ObservationModel* createObservationModel()
128  {
129  RcsBody* plate = RcsGraph_getBodyByName(graph, "Plate");
130  RcsBody* plateOrigMarker = RcsGraph_getBodyByName(graph, "PlateOriginMarker");
131  RCHECK_MSG(plateOrigMarker, "PlateOriginMarker is missing, please update your xml.");
132  // Set origin marker position
133  HTr_copyOrRecreate(&plateOrigMarker->A_BP, plate->A_BI);
134  // Select observation model
135 
136  bool ballObsAngular = properties->getPropertyBool("ballObsAngular");
137 // bool ballObsInInertialFrame = properties->getPropertyBool(
138 // "ballObsInInertialFrame");
139 
140  std::string actionModelType = "joint_pos";
141  properties->getProperty(actionModelType, "actionModelType");
142  bool havePlateLinPos = actionModelType == "plate_acc5d";
143 
144  auto fullState = new OMCombined();
145 
146  // Add plate linear position if desired
147  if (havePlateLinPos) {
148  auto plin = new OMBodyStateLinear(graph, "Plate", "PlateOriginMarker");
149  plin->setMinState(-0.1);
150  plin->setMaxState(0.1);
151  plin->setMaxVelocity(2.0);
152  fullState->addPart(plin);
153  }
154 
155  // Plate angular position
156  auto pang = new OMBodyStateAngular(graph, "Plate", "PlateOriginMarker");
157  pang->setMinState(-45*M_PI/180);
158  pang->setMaxState(45*M_PI/180);
159  pang->setMaxVelocity(2*360*M_PI/180);
160  // Mask out Z since it's fixed
161  fullState->addPart(OMPartial::fromMask(pang, {true, true, false}));
162 
163  // Ball linear state
164  fullState->addPart(new OMBallPos(graph));
165 
166  if (ballObsAngular) {
167  // Add angular ball state
168  fullState->addPart(new OMBodyStateAngular(graph, "Ball", "Plate"));
169  }
170  return fullState;
171  }

194  virtual void initViewer(Rcs::Viewer* viewer)
195  {
196  initManipulability = 0;
197  // Compute initial manipulability
198  auto amIK = actionModel->unwrap<ActionModelIK>();
199  if (amIK != NULL) {
200  initManipulability = amIK->getController()->computeManipulabilityCost();
201  }
202 
203 #ifdef GRAPHICS_AVAILABLE
204  // Set camera above plate
205  RcsBody* plate = RcsGraph_getBodyByName(graph, "Plate");
206  RCHECK(plate);
207  std::string cameraView = "side";
208  properties->getProperty(cameraView, "cameraView");
209 
210  // The camera center is the plate center
211  double cameraCenter[3];
212  Vec3d_copy(cameraCenter, plate->A_BI->org);
213 
214  // The camera location - not specified yet
215  double cameraLocation[3];
216  Vec3d_setZero(cameraLocation);
217 
218  // Camera up vector defaults to z
219  double cameraUp[3];
220  Vec3d_setUnitVector(cameraUp, 2);
221 
222  if (cameraView == "topdown") {
223  // The camera looks down vertically
224  Vec3d_copy(cameraLocation, cameraCenter);
225  // The distance between camera and plate is configurable
226  double cameraToPlateDistanceZ = 1.6;
227  properties->getProperty(cameraToPlateDistanceZ, "cameraToPlateDistance");
228  cameraLocation[2] += cameraToPlateDistanceZ;
229  // The plate's y vector should point up on the view
230  Vec3d_setUnitVector(cameraUp, 1);
231  Vec3d_transRotateSelf(cameraUp, plate->A_BI->rot);
232 
233  }
234  else if (cameraView == "side") {
235  // Make camera center a bit lower, so we use the screen better
236  cameraCenter[2] -= 0.4;
237  // Shift camera pos in x dir
238  cameraLocation[0] = 2.9;
239  cameraLocation[2] = 1.3;
240 
241  // Rotate to world frame
242  Vec3d_transformSelf(cameraLocation, plate->A_BI);
243 
244  }
245  else {
246  RMSG("Unsupported camera view: %s", cameraView.c_str());
247  return;
248  }
249 
250  // Apply the camera position
251  viewer->setCameraHomePosition(osg::Vec3d(cameraLocation[0], cameraLocation[1], cameraLocation[2]),
252  osg::Vec3d(cameraCenter[0], cameraCenter[1], cameraCenter[2]),
253  osg::Vec3d(cameraUp[0], cameraUp[1], cameraUp[2]));
254 #endif
255  }

257  void getHUDText(
258  std::vector<std::string>& linesOut, double currentTime, const MatNd* obs, const MatNd* currentAction,
259  PhysicsBase* simulator, PhysicsParameterManager* physicsManager, ForceDisturber* forceDisturber) override
260  {
261  // Obtain simulator name
262  const char* simName = "None";
263  if (simulator != NULL) {
264  simName = simulator->getClassName();
265  }
266 
267  linesOut.emplace_back(string_format(
268  "physics engine: %s simulation time: %2.3f s",
269  simName, currentTime));
270 
271  unsigned int numPosCtrlJoints = 0;
272  unsigned int numTrqCtrlJoints = 0;
273  // Iterate over unconstrained joints
274  RCSGRAPH_TRAVERSE_JOINTS(graph) {
275  if (JNT->jacobiIndex != -1) {
276  if (JNT->ctrlType == RCSJOINT_CTRL_TYPE::RCSJOINT_CTRL_POSITION) {
277  numPosCtrlJoints++;
278  }
279  else if (JNT->ctrlType == RCSJOINT_CTRL_TYPE::RCSJOINT_CTRL_TORQUE) {
280  numTrqCtrlJoints++;
281  }
282  }
283  }
284  linesOut.emplace_back(
285  string_format("num joints: %d total, %d pos ctrl, %d trq ctrl", graph->nJ, numPosCtrlJoints,
286  numTrqCtrlJoints));
287 
288  linesOut.emplace_back(string_format(
289  "plate angles: [% 3.2f,% 3.2f] deg",
290  RCS_RAD2DEG(obs->ele[0]), RCS_RAD2DEG(obs->ele[1])));
291 
292  const double* distForce = forceDisturber->getLastForce();
293  linesOut.emplace_back(string_format(
294  "disturbances: [% 3.1f,% 3.1f,% 3.1f] N",
295  distForce[0], distForce[1], distForce[2]));
296 
297  if (physicsManager != NULL) {
298  // Get the parameters that are not stored in the Rcs graph
299  BodyParamInfo* ball_bpi = physicsManager->getBodyInfo("Ball");
300  double* com = ball_bpi->body->Inertia->org;
301  double ball_radius = ball_bpi->body->shape[0]->extents[0];
302  double slip = 0;
303  ball_bpi->material.getDouble("slip", slip);
304 
305  linesOut.emplace_back(string_format(
306  "ball mass: %2.2f kg ball radius: %2.3f cm",
307  ball_bpi->body->m, ball_radius*100));
308 
309  linesOut.emplace_back(string_format(
310  "ball friction: %1.2f ball rolling friction: %1.3f",
311  ball_bpi->material.getFrictionCoefficient(),
312  ball_bpi->material.getRollingFrictionCoefficient()/ball_radius));
313 
314  linesOut.emplace_back(string_format(
315  "ball slip: %3.1f rad/(Ns) CoM offset:[% 2.1f, % 2.1f, % 2.1f] mm",
316  slip, com[0]*1000, com[1]*1000, com[2]*1000));
317  }
318 
319  // Compute manipulability cost if available
320  auto amIK = actionModel->unwrap<ActionModelIK>();
321  if (amIK != NULL) {
322  double manipulability = amIK->getController()->computeManipulabilityCost()/initManipulability;
323  linesOut.emplace_back(string_format(
324  "manipulability: %10.8f ",
325  manipulability));
326  }
327  }
