42 /*! Base class for IK-based action models.
43  *
44  * This class encapsulates a Rcs ControllerBase object as well as an inverse kinematics solver.
45  *
46  * To use this, create a subclass and add your tasks in the constructor. Then, use computeIK(...) to invoke the solver.
47  *
48  * In order to facilitate stable kinematics, the joint space error returned by the IK solver is integrated into the
49  * desired joint states ignoring the current measurements. To faciliate this, the ControllerBase object contains
50  * a copy of the main RcsGraph. Thus, all IK tasks should use this graph.
51  */
52 class ActionModelIK : public ActionModel
53 {
54

57  * @param graph current state graph
58  */
59  explicit ActionModelIK(RcsGraph* graph);
60 
61  /*! Create an IK based action model using a fixed set of tasks.
62  * @param graph current state graph
63  * @param tasks controller tasks to use
64  */
65  explicit ActionModelIK(RcsGraph* graph, std::vector<Task*> tasks);
