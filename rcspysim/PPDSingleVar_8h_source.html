41 /**
42  * Descriptor for a single scalar variable of type T.
43  * The actual parameter name is built to be [lower case body name]_[param name].
44  */
45 template<typename T>
46 class PPDSingleVar : public PhysicsParameterDescriptor
47 {
48 public:
49  /*!
50  * Returns a reference to the variable whihc this descriptor uses.
51  */
52  typedef std::function<T & (BodyParamInfo & )> VariableAccessor;
53 
54  /**
55  * Constructor.
56  * @param name unprefixed name for the parameter
57  * @param modifiedFlag modified flag value to set for BodyParamInfo
58  * @param variableAccessor variable accessor function. Returns a reference to be readable and writable.
59  */
60  PPDSingleVar(std::string name, int modifiedFlag, VariableAccessor variableAccessor) :
61  name(std::move(name)), modifiedFlag(modifiedFlag), variableAccessor(variableAccessor) {}
62 
63  virtual ~PPDSingleVar() = default;
64 
65  virtual void getValues(PropertySink* outValues)
66  {
67  outValues->setProperty(name.c_str(), variableAccessor(*bodyParamInfo));
68  }
69 
70  virtual void setValues(PropertySource* inValues)
71  {
72  // Get value reference
73  T& ref = variableAccessor(*bodyParamInfo);
74 
75  // Try to get from dict
76  bool set = inValues->getProperty(ref, name.c_str());
77 
78  // Set changed flag
79  if (set) {
80  bodyParamInfo->markChanged(modifiedFlag);
81  }
82  }
83 
84 protected:
85  virtual void init(BodyParamInfo* bodyParamInfo)
86  {
87  PhysicsParameterDescriptor::init(bodyParamInfo);
88  name = bodyParamInfo->paramNamePrefix + name;
89  }
90 
91 private:
92  //! Parameter name/key which the init() method will add a body prefix to this
93  std::string name;
94 
95  //! Value to or to BodyParamInfo::modifiedFlag when modified
96  int modifiedFlag;
97 
98  //! Variable accessor function
99  VariableAccessor variableAccessor;
100 };
