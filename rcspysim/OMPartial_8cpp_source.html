44 static IndexList loadIndexList(IndexList input, unsigned int dim, bool exclude, const char* category)
45 {
46  // Verify
47  for (auto idx : input) {
48  if (idx >= dim) {
49  std::ostringstream os;
50  os << (exclude ? "Excluded " : "Selected ") << category << " index " << idx
51  << " is outside of the value dimension "
52  << dim;
53  throw std::invalid_argument(os.str());
54  }
55  }
56 
57  // Invert if needed
58  if (exclude) {
59  IndexList out;
60  for (unsigned int idx = 0; idx < dim; ++idx) {
61  // add if not in index list
62  if (std::find(input.begin(), input.end(), idx) == input.end()) {
63  out.push_back(idx);
64  }
65  }
66  return out;
67  }
68  else {
69  return input;
70  }
71 }

73 static IndexList loadMask(const std::vector<bool>& mask, unsigned int dim, bool exclude, const char* category)
74 {
75  // Verify
76  if (mask.size() != dim) {
77  std::ostringstream os;
78  os << category << " mask size " << mask.size() << " does not match value dimension " << dim;
79  throw std::invalid_argument(os.str());
80  }
81  // Convert to index list
82  IndexList out;
83  for (unsigned int idx = 0; idx < dim; ++idx) {
84  // add true entries if exclude is false, or false entries if exclude is true
85  if (mask[idx] == !exclude) {
86  out.push_back(idx);
87  }
88  }
89  return out;
90 }

133 OMPartial *OMPartial::fromNames(ObservationModel *wrapped, const std::vector<std::string> &names, bool exclude,
134  bool autoSelectVelocity) {
135  IndexList states;
136  IndexList velocities;
137 
138  auto stateNames = wrapped->getStateNames();
139  auto velocityNames = wrapped->getVelocityNames();
140 
141  for (auto& name : names) {
142  // find in state
143  auto stateIter = std::find(stateNames.begin(), stateNames.end(), name);
144  if (stateIter != stateNames.end()) {
145  // add to index list
146  states.push_back(std::distance(stateNames.begin(), stateIter));
147 
148  if (autoSelectVelocity) {
149  // select corresponding velocity
150  auto velIter = std::find(velocityNames.begin(), velocityNames.end(), name+"d");
151  if (velIter != velocityNames.end()) {
152  // add to index list
153  velocities.push_back(std::distance(velocityNames.begin(), velIter));
154  }
155  }
156  } else {
157  // find in velocity
158  auto velIter = std::find(velocityNames.begin(), velocityNames.end(), name);
159  if (velIter != velocityNames.end()) {
160  // add to index list
161  velocities.push_back(std::distance(velocityNames.begin(), velIter));
162  } else {
163  // not found
164  std::ostringstream os;
165  os << (exclude ? "Excluded " : "Selected ") << " name " << name
166  << " was not found in the wrapped observation model";
167  throw std::invalid_argument(os.str());
168  }
169  }
170  }
171  return new OMPartial(wrapped, states, velocities, exclude);
172 }

199 void OMPartial::computeObservation(double* state, double* velocity, const MatNd* currentAction, double dt) const
200 {
201  // Allocate temp storage for full, using matnd for potential stack optimization
202  MatNd* state_full = NULL;
203  MatNd* velocity_full = NULL;
204 
205  MatNd_create2(state_full, wrapped->getStateDim(), 1);
206  MatNd_create2(velocity_full, std::max(wrapped->getVelocityDim(), 1u), 1);
207 
208  // Retrieve from wrapped
209  wrapped->computeObservation(state_full->ele, velocity_full->ele, currentAction, dt);
210 
211  // Apply selection
212  apply(state, state_full->ele, keptStateIndices);
213  apply(velocity, velocity_full->ele, keptVelocityIndices);
214 
215  // Clean up potential allocated memory
216  MatNd_destroy(state_full);
217  MatNd_destroy(velocity_full);
218 }

231

233  wrapped->getLimits(minState_full->ele, maxState_full->ele, maxVelocity_full->ele);
234 
235  // Apply selection
236  apply(minState, minState_full->ele, keptStateIndices);
237  apply(maxState, maxState_full->ele, keptStateIndices);
238  apply(maxVelocity, maxVelocity_full->ele, keptVelocityIndices);
239 
240  // Clean up potential allocated memory
241  MatNd_destroy(minState_full);
242  MatNd_destroy(maxState_full);
243  MatNd_destroy(maxVelocity_full);
