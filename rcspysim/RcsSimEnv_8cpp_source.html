57 Rcs::RcsSimEnv::RcsSimEnv(PropertySource* propertySource)
58 {
59  // Set random seed
60  Math_srand48(0);
61 
62  // Load experiment config
63  config = ExperimentConfig::create(propertySource);
64 
65  // Check if all joints are position controlled for skipping a later inverse dynamics control (compliance)
66  unsigned int numPosCtrlJoints = 0;
67  RCSGRAPH_TRAVERSE_JOINTS(config->graph) {
68  if (JNT->jacobiIndex != -1) {
69  if (JNT->ctrlType == RCSJOINT_CTRL_TYPE::RCSJOINT_CTRL_POSITION) {
70  numPosCtrlJoints++;
71  }
72  }
73  }
74  allJointsPosCtrl = config->graph->nJ == numPosCtrlJoints;
75 
76  // Load physics parameter manager from config. Creates a pointer to config->graph
77  physicsManager = config->createPhysicsParameterManager();
78  // Physics simulator instance is nullptr at the start. It will only be created once reset is called.
79  physicsSim = nullptr;
80 
81  // Create force disturber
82  disturber = config->createForceDisturber();
83 
84  // Load init state setter
85  initStateSetter = config->createInitStateSetter();
86 
87  // Initialize temporary matrices
88  q_ctrl = MatNd_clone(config->graph->q);
89  qd_ctrl = MatNd_clone(config->graph->q_dot);
90  T_ctrl = MatNd_create(config->graph->dof, 1);
91 
92  // Other state-related stuff
93  viewer = nullptr;
94  usePhysicsNode = propertySource->getPropertyBool("usePhysicsNode", false);
95  hud = nullptr;
96  currentObservation = config->observationModel->getSpace()->createValueMatrix();
97  currentAction = config->actionModel->getSpace()->createValueMatrix();
98 // adWidgetHandle = -1;
99 
100  currentStep = 0;
101  currentTime = 0;
102 // lastStepReward = 0;
103 
104  // Initialize transition noise (unused if not explicitly flagged)
105  transitionNoiseBuffer = nullptr;
106  tnbIndex = 0;
107  transitionNoiseIncludeVelocity = propertySource->getPropertyBool("transitionNoiseIncludeVelocity");
108 }

129 MatNd* Rcs::RcsSimEnv::reset(PropertySource* domainParam, const MatNd* initState)
130 {
131  // Set random seed
132  Math_srand48(0);
133 
134  // Lock the graph
135  std::unique_lock<std::mutex> lock(graphLock);
136 
137  // Set graph to initial state
138  RcsGraph_setDefaultState(config->graph);
139  RCSGRAPH_TRAVERSE_SENSORS(config->graph) {
140  MatNd_setZero(SENSOR->rawData);
141  }
142 
143  // Apply the initial state
144  if (initState && initStateSetter != nullptr) {
145  std::string errorMsg;
146  if (!initStateSetter->getSpace()->contains(initState, &errorMsg)) {
147  throw std::invalid_argument("init state space does not contain the init state: " + errorMsg);
148  }
149  initStateSetter->applyInitialState(initState);
150  }
151 
152  // Rebuild physics sim with new parameters
153  // We set it to nullptr before creating the new one since there might be exceptions during parameter loading
154  delete physicsSim;
155  physicsSim = nullptr;
156  physicsSim = physicsManager->createSimulator(domainParam);
157 
158  // Reset control command vectors
159  MatNd_copy(q_ctrl, config->graph->q);
160  MatNd_setZero(qd_ctrl);
161  MatNd_setZero(T_ctrl);
162 
163  // Reset stateful models
164  config->actionModel->reset();
165  config->observationModel->reset();
166  // Reset the physics simulator
167 // physicsSim->reset(); // this causes a GUI error
168  physicsSim->resetTime();
169 
170  // Unlock the graph
171  lock.unlock();
172 
173  // Reset local counters
174  currentStep = 0;
175  currentTime = 0;
176 // lastStepReward = 0;
177 
178  // Return initial observation
179  return config->observationModel->computeObservation(nullptr, config->dt);
180 }

182 MatNd* Rcs::RcsSimEnv::step(const MatNd* action, const MatNd* disturbance)
183 {
184  REXEC(6) {
185  std::cout << "--------------------------------------------------" << std::endl;
186  std::cout << "Iteration " << currentStep << std::endl;
187  std::cout << "--------------------------------------------------" << std::endl;
188  }
189 
190  // Validate actions
191  std::string iaMsg;
192  if (!actionSpace()->checkDimension(action, &iaMsg)) {
193  throw std::invalid_argument("Invalid action - " + iaMsg);
194  }
195 
196  // Store the action for the HUD
197  MatNd_copy(currentAction, action);
198 
199  // Lock the graph
200  std::unique_lock<std::mutex> lock(graphLock);
201 
202  /*-------------------------------------------------------------------------*
203  * Obtain simulator inputs from action model
204  *-------------------------------------------------------------------------*/
205  config->actionModel->computeCommand(q_ctrl, qd_ctrl, T_ctrl, action, config->dt);
206 
207  if (config->checkJointLimits) {
208  std::ostringstream jvMsg;
209  unsigned int numJV = 0;
210  RCSGRAPH_TRAVERSE_JOINTS(config->graph) {
211  // Skip the non-IK joints
212  if (JNT->constrained) {
213  continue;
214  }
215  // Check the remaining joints' limits
216  double qi = MatNd_get2(q_ctrl, JNT->jointIndex, 0);
217  if ((qi < JNT->q_min) || (qi > JNT->q_max)) {
218  numJV++;
219  jvMsg << " " << JNT->name << ": " << qi << " [" << JNT->q_min << ", " << JNT->q_max << "["
220  << std::endl;
221  }
222  }
223  if (numJV > 0) {
224  std::ostringstream os;
225  os << "Detected " << numJV << " joint limit violations: " << std::endl << jvMsg.str();
226  throw JointLimitException(os.str());
227  }
228  }
229 
230  if (disturbance != nullptr && disturber != nullptr) {
231  // Check disturbance shape
232  if (disturbance->m != 3 || disturbance->n != 1) {
233  throw std::invalid_argument("Invalid disturbance");
234  }
235  disturber->apply(physicsSim, disturbance->ele);
236  }
237 
238  /*-------------------------------------------------------------------------*
239  * Inverse dynamics in joint space (compliance control)
240  *-------------------------------------------------------------------------*/
241  if (!allJointsPosCtrl) {
242  Rcs::ControllerBase::computeInvDynJointSpace(T_ctrl, config->graph, q_ctrl, 1000.);
243  }
244 
245  /*-------------------------------------------------------------------------*
246  * Call the physicssimulation and get the new current state
247  *-------------------------------------------------------------------------*/
248  physicsSim->setControlInput(q_ctrl, qd_ctrl, T_ctrl);
249 
250  physicsSim->simulate(config->dt, config->graph);
251 
252  /*-------------------------------------------------------------------------*
253  * Apply transition noise if desired
254  *-------------------------------------------------------------------------*/
255  if (transitionNoiseBuffer != nullptr) {
256  // Get noise set to use for this timestamp
257  MatNd* noise;
258  MatNd_fromStack(noise, getInternalStateDim(), 1);
259  MatNd_getColumn(noise, tnbIndex, transitionNoiseBuffer);
260 
261  if (transitionNoiseIncludeVelocity) {
262  // Apply first part to state and second part to velocity
263  VecNd_addSelf(config->graph->q->ele, noise->ele, config->graph->dof);
264  VecNd_addSelf(config->graph->q_dot->ele, &noise->ele[config->graph->dof], config->graph->dof);
265  }
266  else {
267  // Apply to state values only
268  MatNd_addSelf(config->graph->q, noise);
269  }
270 
271  // Increment buffer index
272  tnbIndex++;
273  if (tnbIndex >= transitionNoiseBuffer->n) {
274  tnbIndex = 0;
275  }
276 
277  // Cleanup
278  MatNd_destroy(noise);
279  }
280 
281  /*-------------------------------------------------------------------------*
282  * Store new physics simulation state in the ExperimentConfig's graph
283  *-------------------------------------------------------------------------*/
284  // Apply forward kinematics
285 // RcsGraph_setState(config->graph, nullptr, config->graph->q_dot); // nullptr leads to using the q vector from the graph
286  RcsGraph_setState(config->graph, config->graph->q, config->graph->q_dot);
287 
288  // Unlock the graph
289  lock.unlock();
290 
291  /*-------------------------------------------------------------------------*
292  * For transition noise, we need to update the state of the physics
293  * simulator to match the state of the graph after applying the noise.
294  *-------------------------------------------------------------------------*/
295  if (transitionNoiseBuffer != nullptr) {
296  RCSGRAPH_TRAVERSE_BODIES(config->graph) {
297  physicsSim->applyTransform(BODY, BODY->A_BI);
298  physicsSim->applyLinearVelocity(BODY, BODY->x_dot);
299  physicsSim->applyAngularVelocity(BODY, BODY->omega);
300  }
301  }
302 
303  // Update step counters
304  currentStep++;
305  currentTime += config->dt;
306 
307  /*-------------------------------------------------------------------------*
308  * Assemble results
309  *-------------------------------------------------------------------------*/
310  // Compute observation
311  MatNd* observation = config->observationModel->computeObservation(action, config->dt);
312  MatNd_copy(currentObservation, observation);
313 
314  /*
315  // compute state for goal monitor
316  MatNd* state;
317  if (config->stateModel == config->observationModel) {
318  // use observation if it is the same
319  state = observation;
320  } else {
321  // get explicitly if not
322  state = config->stateModel->computeObservation(config->dt);
323  }
324 
325  // compute reward
326  auto stepReward = config->goalMonitor->stepReward(state, action);
327  // cleanup temporary state mat if any
328  if (state != observation) {
329  MatNd_destroy(state);
330  }
331 
332  lastStepReward = stepReward.reward;
333  */
334 
335  return observation;
336 }

338 void Rcs::RcsSimEnv::render(std::string mode, bool close)
339 {
340 #ifdef GRAPHICS_AVAILABLE
341  if (close) {
342  // Close the viewer
343  if (viewer) {
344  hud = nullptr;
345  delete viewer;
346  viewer = nullptr;
347  }
348  }
349  else if (mode == "human") {
350  if (!viewer) {
351  // Start viewer without shadows, since the shadow renderer has some issues when multiple simulations
352  // are opened in the same process. You can still enable shadows via the key binding.
353  viewer = new Rcs::Viewer(true, false);
354  auto kc = new Rcs::KeyCatcher();
355 
356  // Make nodes resizable so it can update on physics param changes setting that for relevant nodes only
357  // would be better, but it's not possible right now.
358  // Visualize the IK controller graph (a.k.a. desired graph)
359  auto amIK = config->actionModel->unwrap<ActionModelIK>();
360  if (amIK != nullptr) {
361  Rcs::GraphNode* gDesNode = new Rcs::GraphNode(amIK->getController()->getGraph(), true, false);
362  gDesNode->setGhostMode(true);
363  viewer->add(gDesNode);
364  }
365  if (usePhysicsNode) {
366  Rcs::PhysicsNode* pNode = new Rcs::PhysicsNode(physicsSim, true);
367  viewer->add(pNode);
368  pNode->physicsNd->togglePhysicsModel(); // switch off
369  pNode->physicsNd->toggleGraphicsModel(); // switch on
370  }
371  else {
372  Rcs::GraphNode* gNode = new Rcs::GraphNode(config->graph, true, false);
373  gNode->toggleReferenceFrames();
374  viewer->add(gNode);
375  }
376 
377  hud = new Rcs::HUD();
378  // HUD color from config
379  std::string hudColor;
380  if (config->properties->getProperty(hudColor, "hudColor") && !hudColor.empty()) {
381  hud->setColor(hudColor.c_str());
382  }
383  hud->resize(1024, 100);
384 
385  viewer->setWindowSize(0, 0, 1024, 768);
386  viewer->setUpdateFrequency(50.0);
387  viewer->add(hud);
388  viewer->add(kc);
389  viewer->setBackgroundColor("PEWTER");
390 
391  // Experiment specific settings
392  config->initViewer(viewer);
393 
394  viewer->runInThread(graphLock.native_handle());
395  // viewer->toggleVideoRecording(); // start recording right from the start of the sim
396  }
397  // Update HUD
398  auto hudText = config->getHUDText(
399  currentTime, currentObservation, currentAction, physicsSim, physicsManager, disturber
400  );
401  hud->setText(hudText);
402  }
403 #else
404  // Warn about headless mode
405  static bool warned = false;
406  if (!warned)
407  {
408  warned = true;
409  RMSGS("RcsPySim compiled in headless mode, therefore rendering is not available!");
410  }
411 #endif
412 }
