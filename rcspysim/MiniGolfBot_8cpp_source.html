65 /******************************************************************************
66  * Ctrl-C destructor. Tries to quit gracefully with the first Ctrl-C
67  * press, then just exits.
68  *****************************************************************************/
69 static void quit(int /*sig*/)
70 {
71  static int kHit = 0;
72  runLoop = false;
73  fprintf(stderr, "Trying to exit gracefully - %dst attempt\n", kHit + 1);
74  kHit++;
75 
76  if (kHit == 2) {
77  fprintf(stderr, "Exiting without cleanup\n");
78  exit(0);
79  }
80 }
