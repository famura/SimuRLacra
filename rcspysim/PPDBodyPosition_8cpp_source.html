43 PPDBodyPosition::PPDBodyPosition(const bool includeX, const bool includeY, const bool includeZ)
44 {
45  Vec3d_setZero(initPos);
46  Vec3d_setZero(offset);
47 
48  if (includeX) {
49  addChild(new PPDSingleVar<double>(
50  "pos_offset_x", BodyParamInfo::MOD_POSITION, [this](BodyParamInfo& bpi) -> double& { return (offset[0]); })
51  );
52  }
53  if (includeY) {
54  addChild(new PPDSingleVar<double>(
55  "pos_offset_y", BodyParamInfo::MOD_POSITION, [this](BodyParamInfo& bpi) -> double& { return (offset[1]); })
56  );
57  }
58  if (includeZ) {
59  addChild(new PPDSingleVar<double>(
60  "pos_offset_z", BodyParamInfo::MOD_POSITION, [this](BodyParamInfo& bpi) -> double& { return (offset[2]); })
61  );
62  }
63 
64  if (getChildren().empty()) {
65  throw std::invalid_argument("No position specified for PPDBodyPosition!");
66  }
67 }
